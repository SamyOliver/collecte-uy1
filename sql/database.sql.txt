-- Indiquer explicitement la base de données à utiliser sur Aiven
USE defaultdb;

-- 1. Création de la table des utilisateurs
CREATE TABLE IF NOT EXISTS utilisateurs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nom_utilisateur VARCHAR(50) NOT NULL UNIQUE,
    mot_de_passe VARCHAR(255) NOT NULL,
    role ENUM('admin', 'collecteur') DEFAULT 'collecteur',
    date_creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Création de la table des questions
CREATE TABLE IF NOT EXISTS questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    libelle_question TEXT NOT NULL,
    type_reponse ENUM('texte', 'nombre', 'choix_multiple') DEFAULT 'texte'
);

-- 3. Création de la table des réponses (Adaptée au formulaire UY1)
CREATE TABLE IF NOT EXISTS reponses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    contenu_reponse TEXT,
    collecteur_id INT,
    sexe VARCHAR(20),
    niveau VARCHAR(50),
    etablissement VARCHAR(100),
    restaurant VARCHAR(100),
    note_qualite INT,
    date_reponse TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_question FOREIGN KEY (question_id) REFERENCES questions(id) ON DELETE CASCADE
);
